  • Exitcodes prüfen und Fehlerpfade behandeln.

Grundlagen

# if / elif / else
if [[ -f "$1" ]]; then
  echo "Datei existiert"
elif [[ -d "$1" ]]; then
  echo "Verzeichnis"
else
  echo "Unbekannt"; exit 1
fi

# case
case "$2" in
  start) echo start ;; 
  stop)  echo stop ;;
  *)     echo "Usage: $0  {start|stop}"; exit 2 ;;
esac

# for / while
for f in *.conf; do echo "$f"; done
while read -r line; do printf '%s\n' "$line"; done < "$1"

# Arithmetik
count=0; ((count++)); echo "$count"

# Exitcodes prüfen
cmd || { echo "cmd fehlgeschlagen" >&2; exit 1; }

Cheatsheet

[[ -n "$s" ]]         # String nicht leer
[[ "$a" == foo* ]]    # Pattern
[[ $x -gt 3 ]]         # integer Vergleich
[[ -e path ]]          # Datei/Verz existiert
(( x += 2 ))           # Arithmetik

  3. Validiere drei Integer-Parameter und gib die Summe zurück (inkl. Fehlerpfade).
