Beispielstruktur

#!/usr/bin/env bash
set -Eeuo pipefail

usage() { echo "Usage: $0 -n NAME"; }

log()   { printf '[%s] %s\n' "$(date +%T)" "$*"; }
err()   { echo "ERROR: $*" >&2; }

greet() {
  local name=${1:?name required}
  echo "Hallo $name"
}

main() {
  local name=""
  while getopts ':n:h' opt; do
    case $opt in
      n) name=$OPTARG ;;
      h) usage; exit 0 ;;
      *) usage; exit 2 ;;
    esac
  done
  [[ -n $name ]] || { usage; exit 2; }
  greet "$name"
}

main "$@"
