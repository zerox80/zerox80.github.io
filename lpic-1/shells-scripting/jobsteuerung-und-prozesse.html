Grundlagen

# Jobsteuerung
sleep 100 &            # im Hintergrund
jobs -l                # Jobs auflisten
fg %1; bg %1           # in Vorder-/Hintergrund

# Signale
trap 'echo "abbruch"; exit 2' INT TERM
kill -TERM 1234        # freundlich beenden
kill -KILL 1234        # hart beenden

# Prozesse finden
pgrep -af nginx
ps -o pid,ppid,cmd -p 1
pstree -a -p 1
